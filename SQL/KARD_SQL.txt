SELECT SUM(AMOUNT*0.1) AS TOTAL_DISCOUNT, CLIENT.NAME, CLIENT.SURNAME
FROM ORDERS, CLIENT
WHERE CLIENT.ID = ORDERS.CLIENT_ID AND
MONTH(ORD_TIME) = MONTH(BIRTHDAY) AND
DAY(ORD_TIME) = DAY(BIRTHDAY) AND
YEAR(ORD_TIME) = 2022
GROUP BY CLIENT.ID

SELECT FIRM.NAME, FIRM.ADDRESS, SUM(AMOUNT) AS TOTAL_AMONUT
FROM ORDERS, FIRM, CLIENT
WHERE ORDERS.CLIENT_ID = CLIENT.ID AND
FIRM.ID = CLIENT.FIRM_ID
AND YEAR(ORD_TIME) < 2023
GROUP BY FIRM.ID
HAVING TOTAL_AMONUT > 1000000

SELECT CLIENT.ID, CLIENT.SURNAME, CLIENT.NAME, CLIENT.BIRTHDAY
FROM CLIENT, (
  SELECT NAME, SURNAME, COUNT(ID) FROM CLIENT
  WHERE GENDER = 'M'
  GROUP BY SURNAME
  HAVING COUNT(ID) > 1
) AS COMMON_SURNAMES
WHERE CLIENT.SURNAME = COMMON_SURNAMES.SURNAME

SELECT *
FROM 
(
SELECT FIRM.NAME, SUM(AMOUNT) AS TOTAL_AMOUNT
    FROM FIRM, ORDERS, CLIENT
    WHERE ORDERS.CLIENT_ID = CLIENT.ID AND
    FIRM.ID = CLIENT.FIRM_ID AND
    YEAR(ORD_TIME) = 2022 AND
    MONTH(ORD_TIME) = 12
    GROUP BY FIRM_ID
    
UNION
    
SELECT CONCAT(CLIENT.SURNAME, ' ', CLIENT.NAME) AS NAME, SUM(AMOUNT) AS TOTAL_AMOUNT
    FROM FIRM, ORDERS, CLIENT
    WHERE ORDERS.CLIENT_ID = CLIENT.ID AND
    FIRM_ID IS NULL AND
    YEAR(ORD_TIME) = 2022 AND
    MONTH(ORD_TIME) = 12
    GROUP BY CLIENT.ID
) AS DATA
ORDER BY DATA.TOTAL_AMOUNT DESC
LIMIT 3
